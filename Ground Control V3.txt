// 5. Tasks/Calendar View
const TasksView = ({ tasks, deals, user }) => {
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [newTask, setNewTask] = useState({ title: '', dueDate: '', dealId: '', notes: '' });

  const handleAddTask = async () => {
    if (!newTask.title) return;
    try {
      await addDoc(getAppCollection(db, 'tasks'), {
        ...newTask,
        completed: false,
        createdAt: serverTimestamp()
      });
      setNewTask({ title: '', dueDate: '', dealId: '', notes: '' });
      setIsModalOpen(false);
    } catch (error) {
      console.error("Error adding task:", error);
    }
  };

  const toggleTask = async (task) => {
    try {
      const taskRef = doc(getAppCollection(db, 'tasks'), task.id);
      await updateDoc(taskRef, { completed: !task.completed });
    } catch (error) {
      console.error("Error toggling task:", error);
    }
  };

  const deleteTask = async (taskId) => {
    try {
      await deleteDoc(doc(getAppCollection(db, 'tasks'), taskId));
    } catch (error) {
      console.error("Error deleting task:", error);
    }
  };

  // Sort tasks: Incomplete first, then by date
  const sortedTasks = useMemo(() => {
    return [...tasks].sort((a, b) => {
      if (a.completed === b.completed) {
        return new Date(a.dueDate || '9999-12-31') - new Date(b.dueDate || '9999-12-31');
      }
      return a.completed ? 1 : -1;
    });
  }, [tasks]);

  const getDealName = (dealId) => {
    const deal = deals.find(d => d.id === dealId);
    return deal ? deal.name : 'General Task';
  };

  const formatDate = (dateString) => {
    if (!dateString) return 'No Date';
    return new Date(dateString).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
  };

  const isOverdue = (dateString) => {
    if (!dateString) return false;
    const today = new Date();
    today.setHours(0,0,0,0);
    const due = new Date(dateString);
    return due < today;
  };

  return (
    <div className="h-full flex flex-col">
      <div className="flex justify-between items-center mb-6">
        <div>
          <h2 className="text-2xl font-bold text-slate-800">Critical Dates & Tasks</h2>
          <p className="text-slate-500">Track deadlines, due diligence periods, and action items.</p>
        </div>
        <button 
          onClick={() => setIsModalOpen(true)}
          className="bg-slate-900 text-white px-4 py-2 rounded-lg flex items-center gap-2 hover:bg-slate-700"
        >
          <Plus size={18} /> New Task
        </button>
      </div>

      <div className="bg-white rounded-xl shadow-sm border border-slate-200 overflow-hidden flex-1 flex flex-col">
        <div className="overflow-y-auto flex-1">
          {sortedTasks.length === 0 ? (
            <div className="flex flex-col items-center justify-center h-64 text-slate-400">
              <CheckSquare size={48} className="mb-4 opacity-50" />
              <p>No active tasks. Good job!</p>
            </div>
          ) : (
            <table className="w-full text-left border-collapse">
              <thead className="bg-slate-50 text-slate-500 text-xs uppercase tracking-wider sticky top-0 z-10">
                <tr>
                  <th className="p-4 w-12 text-center">Done</th>
                  <th className="p-4">Task / Action</th>
                  <th className="p-4">Related Deal</th>
                  <th className="p-4">Due Date</th>
                  <th className="p-4 w-12"></th>
                </tr>
              </thead>
              <tbody className="divide-y divide-slate-100">
                {sortedTasks.map(task => (
                  <tr key={task.id} className={`group hover:bg-slate-50 transition ${task.completed ? 'bg-slate-50/50' : ''}`}>
                    <td className="p-4 text-center">
                      <button 
                        onClick={() => toggleTask(task)}
                        className={`w-6 h-6 rounded border flex items-center justify-center transition-colors ${
                          task.completed 
                            ? 'bg-emerald-500 border-emerald-500 text-white' 
                            : 'border-slate-300 hover:border-emerald-500'
                        }`}
                      >
                        {task.completed && <CheckSquare size={14} />}
                      </button>
                    </td>
                    <td className="p-4">
                      <p className={`font-medium text-slate-800 ${task.completed ? 'line-through text-slate-400' : ''}`}>
                        {task.title}
                      </p>
                      {task.notes && <p className="text-xs text-slate-500 mt-1">{task.notes}</p>}
                    </td>
                    <td className="p-4">
                      <span className="inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium bg-blue-100 text-blue-800">
                        {getDealName(task.dealId)}
                      </span>
                    </td>
                    <td className="p-4">
                      <div className={`flex items-center gap-2 text-sm ${
                        !task.completed && isOverdue(task.dueDate) ? 'text-red-600 font-bold' : 'text-slate-600'
                      }`}>
                        <Clock size={14} />
                        {formatDate(task.dueDate)}
                        {!task.completed && isOverdue(task.dueDate) && <span className="text-[10px] bg-red-100 text-red-600 px-1 rounded uppercase">Overdue</span>}
                      </div>
                    </td>
                    <td className="p-4 text-right">
                      <button onClick={() => deleteTask(task.id)} className="text-slate-300 hover:text-red-500 opacity-0 group-hover:opacity-100 transition">
                        <Trash2 size={16} />
                      </button>
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          )}
        </div>
      </div>

      <Modal isOpen={isModalOpen} onClose={() => setIsModalOpen(false)} title="New Task / Critical Date">
        <div className="space-y-4">
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">What needs to be done?</label>
            <input 
              type="text" 
              placeholder="e.g. Sign Purchase Agreement, File Zoning Permit"
              className="w-full p-2 border rounded-lg focus:ring-2 focus:ring-emerald-500 outline-none"
              value={newTask.title}
              onChange={e => setNewTask({...newTask, title: e.target.value})}
            />
          </div>
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">Due Date / Deadline</label>
            <input 
              type="date" 
              className="w-full p-2 border rounded-lg focus:ring-2 focus:ring-emerald-500 outline-none"
              value={newTask.dueDate}
              onChange={e => setNewTask({...newTask, dueDate: e.target.value})}
            />
          </div>
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">Related Deal</label>
            <select 
              className="w-full p-2 border rounded-lg focus:ring-2 focus:ring-emerald-500 outline-none"
              value={newTask.dealId}
              onChange={e => setNewTask({...newTask, dealId: e.target.value})}
            >
              <option value="">-- General Task --</option>
              {deals.map(d => (
                <option key={d.id} value={d.id}>{d.name}</option>
              ))}
            </select>
          </div>
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">Notes</label>
            <textarea 
              className="w-full p-2 border rounded-lg focus:ring-2 focus:ring-emerald-500 outline-none"
              rows="3"
              value={newTask.notes}
              onChange={e => setNewTask({...newTask, notes: e.target.value})}
            ></textarea>
          </div>
          <button 
            onClick={handleAddTask}
            className="w-full bg-emerald-600 text-white py-2 rounded-lg font-medium hover:bg-emerald-700 mt-4"
          >
            Save Task
          </button>
        </div>
      </Modal>
    </div>
  );
};

// 6. Main App Component
const App = () => {
  const [user, setUser] = useState(null);
  const [activeView, setActiveView] = useState('pipeline');
  const [deals, setDeals] = useState([]);
  const [contacts, setContacts] = useState([]);
  const [tasks, setTasks] = useState([]);

  // Auth
  useEffect(() => {
    const initAuth = async () => {
      // Logic for Canvas Auth (uses custom token) vs Standalone Auth (anonymous fallback)
      if (isCanvasEnvironment && typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  // Data Sync
  useEffect(() => {
    if (!user) return;

    // Deals Listener
    const dealsUnsub = onSnapshot(
      getAppCollection(db, 'deals'),
      (snapshot) => {
        setDeals(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
      },
      (error) => console.error("Deals sync error:", error)
    );

    // Contacts Listener
    const contactsUnsub = onSnapshot(
      getAppCollection(db, 'contacts'),
      (snapshot) => {
        setContacts(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
      },
      (error) => console.error("Contacts sync error:", error)
    );

    // Tasks Listener
    const tasksUnsub = onSnapshot(
      getAppCollection(db, 'tasks'),
      (snapshot) => {
        setTasks(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
      },
      (error) => console.error("Tasks sync error:", error)
    );

    return () => {
      dealsUnsub();
      contactsUnsub();
      tasksUnsub();
    };
  }, [user]);

  if (!user) return <div className="flex items-center justify-center h-screen text-slate-500">Loading GroundControl...</div>;

  return (
    <div className="flex h-screen bg-slate-100 font-sans text-slate-900">
      <Sidebar activeView={activeView} setActiveView={setActiveView} />
      <main className="flex-1 ml-64 p-8 overflow-hidden h-screen">
        {activeView === 'pipeline' && <PipelineView deals={deals} setDeals={setDeals} user={user} />}
        {activeView === 'contacts' && <ContactsView contacts={contacts} deals={deals} user={user} />}
        {activeView === 'tasks' && <TasksView tasks={tasks} deals={deals} user={user} />}
      </main>
    </div>
  );
};

export default App;
